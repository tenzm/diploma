\structure{ПЕРЕЧЕНЬ СОКРАЩЕНИЙ И ОБОЗНАЧЕНИЙ}
\addcontentsline{toc}{structure}{Перечень сокращений и обозначений}

\begin{tabular}{lp{0.75\textwidth}}
\textbf{ML}  & Machine Learning --- машинное обучение \\
\textbf{LLM} & Large Language Model --- большая языковая модель \\
\textbf{SLA} & Service Level Agreement --- соглашение об уровне обслуживания \\
\textbf{SLO} & Service Level Objective --- целевой показатель уровня обслуживания \\
\textbf{K8s} & Kubernetes --- система оркестрации контейнеров \\
\textbf{P2P} & Peer-to-Peer --- одноранговая (децентрализованная) сеть \\
\textbf{FUSE} & Filesystem in Userspace --- файловая система в пространстве пользователя \\
\textbf{UDS} & Unix Domain Socket --- сокет домена Unix \\
\textbf{TCP} & Transmission Control Protocol --- протокол управления передачей \\
\textbf{NFS} & Network File System --- сетевая файловая система \\
\textbf{PV}  & PersistentVolume --- постоянный том Kubernetes \\
\textbf{PVC} & PersistentVolumeClaim --- заявка на постоянный том Kubernetes \\
\textbf{RWX} & ReadWriteMany --- режим доступа к тому Kubernetes (чтение и запись несколькими нодами) \\
\textbf{API} & Application Programming Interface --- программный интерфейс приложения \\
\textbf{RTT} & Round-Trip Time --- время двойного прохода сигнала \\
\textbf{BDP} & Bandwidth-Delay Product --- произведение полосы пропускания на задержку \\
\textbf{IPC} & Inter-Process Communication --- межпроцессное взаимодействие \\
\textbf{VFS} & Virtual File System --- виртуальная файловая система \\
\textbf{LRU} & Least Recently Used --- алгоритм вытеснения наименее недавно используемых данных \\
\textbf{TTL} & Time-To-Live --- время жизни записи в кэше или реестре \\
\textbf{HA}  & High Availability --- высокая доступность \\
\textbf{SPOF} & Single Point of Failure --- единственная точка отказа \\
\textbf{CDN} & Content Delivery Network --- сеть доставки содержимого \\
\textbf{CNN} & Convolutional Neural Network --- сверточная нейронная сеть \\
\textbf{RNN} & Recurrent Neural Network --- рекуррентная нейронная сеть \\
\textbf{GPU} & Graphics Processing Unit --- графический процессор \\
\textbf{CPU} & Central Processing Unit --- центральный процессор \\
\textbf{AMAT} & Average Memory Access Time --- среднее время доступа к памяти \\
\textbf{mDNS} & Multicast DNS --- многоадресный DNS \\
\textbf{CNI} & Container Network Interface --- сетевой интерфейс контейнеров \\
\textbf{CSI} & Container Storage Interface --- интерфейс хранилища контейнеров \\
\end{tabular}
